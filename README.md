    <script>
        // Check authentication status on page load
        document.addEventListener('DOMContentLoaded', function() {
            const isRealUser = localStorage.getItem('isRealUser') === 'true';
            const isAuthenticated = localStorage.getItem('isAuthenticated') === 'true';
            
            if (isAuthenticated) {
                document.getElementById('auth-section').style.display = 'none';
                document.getElementById('homepage-container').style.display = 'block';
                
                if (!isRealUser) {
                    document.getElementById('access-warning').classList.remove('hidden');
                    restrictLinks();
                }
            }
        });

        // Toggle between sign-in and sign-up forms
        document.getElementById('show-signin').addEventListener('click', function(event) {
            event.preventDefault();
            document.getElementById('signup-form').classList.add('hidden');
            document.getElementById('signin-form').classList.remove('hidden');
            document.getElementById('error-message').style.display = 'none';
        });

        document.getElementById('show-signup').addEventListener('click', function(event) {
            event.preventDefault();
            document.getElementById('signin-form').classList.add('hidden');
            document.getElementById('signup-form').classList.remove('hidden');
            document.getElementById('error-message').style.display = 'none';
        });

        // Handle sign-in form submission
        document.getElementById('signin-form').addEventListener('submit', function(event) {
            event.preventDefault();
            
            const username = document.getElementById('signin-username').value;
            const password = document.getElementById('signin-password').value;
            
            if (username === "johnmichael" && password === "johnmichael") {
                localStorage.setItem('isAuthenticated', 'true');
                localStorage.setItem('isRealUser', 'true');
                
                document.getElementById('auth-section').style.display = 'none';
                document.getElementById('homepage-container').style.display = 'block';
                document.getElementById('error-message').style.display = 'none';
            } else {
                document.getElementById('error-message').style.display = 'block';
                document.getElementById('signin-password').value = '';
            }
        });

        // Handle sign-up form submission
        document.getElementById('signup-form').addEventListener('submit', function(event) {
            event.preventDefault();
            
            const username = document.getElementById('signup-username').value;
            const email = document.getElementById('signup-email').value;
            const password = document.getElementById('signup-password').value;
            
            if (username && email && password) {
                localStorage.setItem('isAuthenticated', 'true');
                localStorage.setItem('isRealUser', 'false');
                
                document.getElementById('auth-section').style.display = 'none';
                document.getElementById('homepage-container').style.display = 'block';
                document.getElementById('access-warning').classList.remove('hidden');
                restrictLinks();
            } else {
                alert("Please fill in all fields!");
            }
        });

        // Function to restrict links for non-real users
        function restrictLinks() {
            const links = document.querySelectorAll('.menu a');
            links.forEach(link => {
                link.classList.add('restricted');
                link.addEventListener('click', function(event) {
                    event.preventDefault();
                    alert("Access Restricted! Only real users with valid credentials can access this page.\n\nPlease sign in");
                });
            });
        }

        // Logout functionality
        document.getElementById('logout-btn').addEventListener('click', function() {
            localStorage.removeItem('isAuthenticated');
            localStorage.removeItem('isRealUser');
            
            document.getElementById('homepage-container').style.display = 'none';
            document.getElementById('auth-section').style.display = 'block';
            document.getElementById('signin-form').classList.remove('hidden');
            document.getElementById('signup-form').classList.add('hidden');
            document.getElementById('error-message').style.display = 'none';
            document.getElementById('access-warning').classList.add('hidden');
            
            document.getElementById('signin-form').reset();
            document.getElementById('signup-form').reset();
        });
    </script>
